# 1. 先检查 virtualenv 是否正确安装（验证安装结果）
virtualenv --version  # 若输出版本号（如 20.34.0），说明安装成功

# 2. 问题原因：mkvirtualenv 是 virtualenvwrapper 工具的命令，而非 virtualenv 自带
# 解决方案1：用 virtualenv 原生命令创建并激活虚拟环境（推荐，无需额外安装）
# 创建名为 shop_env 的虚拟环境（会在当前目录生成 shop_env 文件夹）
virtualenv shop_env

# 激活虚拟环境（macOS/Linux 系统）
source shop_env/bin/activate

# 激活后终端前缀会显示 (shop_env)，表示激活成功
# 后续安装依赖（如 flask）会自动安装到该虚拟环境中

# 查看端口占用 sudo lsof -i tcp:8088
# 终止进程 sudo kill -9 PID
